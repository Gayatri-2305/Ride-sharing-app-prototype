    script>
        document.addEventListener('DOMContentLoaded', function() {
            // Select ride option
            const rideOptions = document.querySelectorAll('.ride-option');
            rideOptions.forEach(option => {
                option.addEventListener('click', function() {
                    rideOptions.forEach(opt => opt.classList.remove('selected'));
                    this.classList.add('selected');
                });
            });
            
            // Request ride button
            const requestRideBtn = document.getElementById('request-ride-btn');
            const driverMatchingModal = document.getElementById('driver-matching-modal');
            const searchingDriver = document.getElementById('searching-driver');
            const driverMatched = document.getElementById('driver-matched');
            
            requestRideBtn.addEventListener('click', function() {
                driverMatchingModal.classList.remove('hidden');
                requestRideBtn.classList.remove('pulse');
                
                // Simulate driver search
                setTimeout(() => {
                    searchingDriver.classList.add('hidden');
                    driverMatched.classList.remove('hidden');
                    driverMatched.classList.add('show');
                }, 3000);
            });
            
            // Cancel ride button
            const cancelRideBtn = document.getElementById('cancel-ride-btn');
            cancelRideBtn.addEventListener('click', function() {
                driverMatchingModal.classList.add('hidden');
                searchingDriver.classList.remove('hidden');
                driverMatched.classList.add('hidden');
                driverMatched.classList.remove('show');
                requestRideBtn.classList.add('pulse');
            });
            
            // Current location button
            const currentLocationBtn = document.getElementById('current-location-btn');
            currentLocationBtn.addEventListener('click', function() {
                alert('Finding your current location...');
            });
        });
    /script>
